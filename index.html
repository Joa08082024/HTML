document.getElementById("freteiroForm").addEventListener("submit", function(event) { event.preventDefault(); const codMotorista = parseInt(document.getElementById("codMotorista").value); const codMotivo = parseInt(document.getElementById("codMotivo").value); const numeroNota = document.getElementById("numeroNota").value; const valor = parseFloat(document.getElementById("valor").value); // Validar entrada if (isNaN(codMotorista) || isNaN(codMotivo) || isNaN(valor) || !numeroNota.trim()) { alert("Por favor, preencha todos os campos corretamente."); return; } const freteiroNome = freteiros[codMotorista] || "Desconhecido"; const motivoNome = motivos[codMotivo] || "Desconhecido"; // Monta o objeto a ser enviado const freteiroData = { codMotorista, codMotivo, numeroNota, valor, freteiroNome, motivoNome }; // Envia os dados para o servidor fetch('/add-freteiro', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(freteiroData) }) .then(response => response.text()) .then(data => { alert(data); // Mostra uma mensagem de sucesso // Adiciona na tabela como antes const tabela = document.getElementById("tabelaFreteiros").getElementsByTagName("tbody")[0]; const novaLinha = tabela.insertRow(); novaLinha.insertCell(0).textContent = tabela.rows.length; // Número da linha novaLinha.insertCell(1).textContent = codMotorista; novaLinha.insertCell(2).textContent = freteiroNome; novaLinha.insertCell(3).textContent = motivoNome; novaLinha.insertCell(4).textContent = numeroNota; novaLinha.insertCell(5).textContent = valor.toFixed(2); total += valor; // Atualizar o total document.getElementById("total").textContent = total.toFixed(2); // Exibir o total atualizado // Limpar formulário document.getElementById("freteiroForm").reset(); }) .catch(error => { console.error('Erro:', error); alert('Erro ao cadastrar freteiro.'); }); });
